// -------- Existing Game Logic --------														
const pool = [														
{name:'Iron Fragment', displayQty:'x4000', weight:8},														
{name:'Stone', displayQty:'x10000', weight:7},														
{name:'Sulfur', displayQty:'x4000', weight:6},														
{name:'Gun Barrel', displayQty:'x30', weight:5},														
{name:'Musket', displayQty:'x10', weight:4},														
{name:'Pistol', displayQty:'x20', weight:4},														
{name:'Shotgun', displayQty:'x10', weight:3},														
{name:'3 Month Tribals+', displayQty:'x1', weight:2}														
];														
														
function weightedRandom(){														
const total = pool.reduce((s,p)=>s+p.weight,0);														
let r = Math.random()*total;														
for(const p of pool){ if(r < p.weight) return p; r -= p.weight; }														
return pool[0];														
}														
														
function startTrade(){														
const tradeBox = document.getElementById('tradeBox');														
const status = document.getElementById('status');														
tradeBox.innerHTML = '';														
status.textContent = 'Rolling...';														
														
const slots = [];														
for(let i=0;i<9;i++){														
const pick = weightedRandom();														
const el = document.createElement('div'); el.className='item';														
el.textContent = `${pick.name} ${pick.displayQty}`;														
slots.push({el,pick}); tradeBox.appendChild(el);														
}														
														
let i = 0; let speed = 80; let cycles = 24;														
const interval = setInterval(()=>{														
slots.forEach(s=>s.el.style.transform='scale(1)');														
slots[i%slots.length].el.style.transform='scale(1.08)';														
i++; if(i>cycles){ clearInterval(interval);														
const winnerIndex = Math.floor(Math.random()*slots.length);														
slots.forEach((s,idx)=>{														
s.el.style.opacity = idx===winnerIndex ? '1' : '0.35';														
s.el.style.transform = idx===winnerIndex ? 'scale(1.12)' : 'scale(1)';														
});														
status.textContent = 'ðŸŽ‰ Winner: ' + slots[winnerIndex].pick.name + ' ' + slots[winnerIndex].pick.displayQty + '!';														
}														
}, speed);														
}
